This is an example text where you can click on every word to highlight it.

/* Add styles as needed */ .text-container { font-size: 18px; } .highlighted { background-color: yellow; cursor: pointer; } const textContainer = document.getElementById('text-container'); textContainer.addEventListener('click', (event) => { if (event.target.nodeName === 'P') { const words = event.target.textContent.split(' '); event.target.innerHTML = words.map(word => `${word}`).join(' '); } else if (event.target.classList.contains('highlighted')) { event.target.classList.remove('highlighted'); } });
